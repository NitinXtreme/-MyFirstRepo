// server.js const express = require("express"); const bodyParser = require("body-parser"); const cors = require("cors"); const app = express(); const PORT = 3000; app.use(cors()); app.use(bodyParser.urlencoded({ extended: false })); app.use(bodyParser.json()); app.post("/feedback", (req, res) => { const { name, email, message } = req.body; console.log("Feedback received:"); console.log("Name:", name); console.log("Email:", email); console.log("Message:", message); res.send("Feedback submitted successfully!"); }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
